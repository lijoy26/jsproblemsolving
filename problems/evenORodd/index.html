• Odd numbers are whole number wchich can't divide by 2 eg:- 1,3,5,7..
  • Even numbers are numbers wchich can divide by 2 witout any mode eg:- 0,2,4,6..
This code will illustrate check number even or odd without having a conditional operation.
function checknumber() {
 let evenorodd = ['Even', 'Odd'];
 let number = document.querySelector('#js-number-input').value;
 document.querySelector('#js-result').innerHTML = evenorodd[number % 2];
}
